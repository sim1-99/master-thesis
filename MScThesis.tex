\def\numberline#1{%
  \ifx\Hy@tocdestname\ltx@empty
    \hb@xt@\@tempdima{#1\hfill}%
  \else
  \hb@xt@\@tempdima{
    \hyper@linkstart{link}{\Hy@tocdestname}#1\hyper@linkend\hfill
  }%
 \fi
}
